function addTask(){
    let input = document.getElementById("taskInput");
    let text = input.value.trim();
    if(!text) return;

    let li = document.createElement("li");
    li.innerHTML = `<span>${text}</span>
        <div class="actions">
            <button onclick="this.closest('li').querySelector('span').classList.toggle('done')">Done</button>
            <button onclick="this.closest('li').remove()">Delete</button>
        </div>`;
    document.getElementById("taskList").appendChild(li);
    input.value = "";
}
</script>
Q2 — To-Do List Part 2 (localStorage) 
<!DOCTYPE html>
<html>
<head>
  <title>To-Do List</title>
</head>
<body>
<h3>My Day</h3>
<input type="text" id="in" placeholder="Enter task">
<button onclick="add()">Add</button>
<ul id="list"></ul>
<button onclick="clearAll()">Clear All</button>
<script>
  let tasks = JSON.parse(localStorage.getItem("tasks")) || [];
  function add() {
    let val = document.getElementById("in").value;
    if (!val.trim()) return;

    tasks.unshift({ id: Date.now(), text: val, done: false });
    render();
  }
  function toggle(id) {
    tasks = tasks.map(t =>
      t.id === id ? { ...t, done: !t.done } : t
    );
    render();
  }
  function del(id) {
    tasks = tasks.filter(t => t.id !== id);
    render();
  }
  function clearAll() {
    tasks = [];
    render();
  }
  function render() {
    localStorage.setItem("tasks", JSON.stringify(tasks));
    let list = document.getElementById("list");
    list.innerHTML = "";
    tasks.forEach(t => {
      let li = document.createElement("li");
      li.innerHTML = `
        <span onclick="toggle(${t.id})"
              style="${t.done ? 'text-decoration:line-through' : ''}">
          ${t.text}
        </span>
        <button onclick="del(${t.id})">X</button>
      `;
      list.appendChild(li);
    });
    document.getElementById("in").value = "";
  }
  render();
</script>
</body>
</html>

Q8 — React Counter (useState + useEffect) 
import React, { useState, useEffect } from "react";

function Counter() {
    const [count, setCount] = useState(0);

    useEffect(() => {
        document.title = `Count: ${count}`;
    }, [count]);

    return (
        <div style={{textAlign:"center", marginTop:"40px"}}>
            <h2>Counter App</h2>
            <h3>{count}</h3>

            <button onClick={() => setCount(count + 1)}>+</button>
            <button onClick={() => setCount(count - 1)} style={{marginLeft:"10px"}}>-</button>
            <button onClick={() => setCount(0)} style={{marginLeft:"10px"}}>Reset</button>
        </div>
    );
}

export default Counter;
Q5 — Weather App (AJAX + API) 
<!DOCTYPE html>
<html>
<head>
  <title>Weather App</title>
</head>
<body>
<h2>Weather App</h2>
<input type="text" id="city" placeholder="Enter city">
<button onclick="getWeather()">Get Weather</button>
<div id="result"></div>

<script>
const API_KEY = "your_api_key_here";

function getWeather() {
  let city = document.getElementById("city").value.trim();
  if (!city) return;

  let url = `https://api.openweathermap.org/data/2.5/weather?q=${city}&appid=${API_KEY}&units=metric`;

  fetch(url)
    .then(res => res.json())
    .then(data => {
      if (data.cod !== 200) {
        document.getElementById("result").innerHTML = "<p>City not found.</p>";
        return;
      }
      document.getElementById("result").innerHTML = `
        <h3>${data.name}, ${data.sys.country}</h3>
        <p>Temperature: ${data.main.temp} °C</p>
        <p>Humidity: ${data.main.humidity}%</p>
        <p>Condition: ${data.weather[0].description}</p>
      `;
    })
    .catch(err => {
      document.getElementById("result").innerHTML = "<p>Error fetching data.</p>";
    });
}
</script>
</body>
</html>

Q9 — Search Filter on API Results 
<!DOCTYPE html>
<html>
<head>
  <title>Search Filter</title>
  <style>
    body { font-family: Arial; padding: 20px; }
    input { padding: 8px; width: 300px; margin-bottom: 15px; }
    .card { border: 1px solid #ccc; border-radius: 8px; padding: 10px; margin: 8px 0; }
  </style>
</head>
<body>
<h2>User Search</h2>
<input type="text" id="search" placeholder="Search by name..." oninput="filterUsers()">
<div id="output">Loading...</div>

<script>
let allUsers = [];

fetch("https://jsonplaceholder.typicode.com/users")
  .then(res => res.json())
  .then(data => {
    allUsers = data;
    renderUsers(allUsers);
  });

function filterUsers() {
  let query = document.getElementById("search").value.toLowerCase();
  let filtered = allUsers.filter(u => u.name.toLowerCase().includes(query));
  renderUsers(filtered);
}

function renderUsers(users) {
  let output = document.getElementById("output");
  if (users.length === 0) {
    output.innerHTML = "<p>No users found.</p>";
    return;
  }
  output.innerHTML = users.map(u => `
    <div class="card">
      <strong>${u.name}</strong><br>
      Email: ${u.email}<br>
      City: ${u.address.city}
    </div>
  `).join("");
}
</script>
</body>
</html>
